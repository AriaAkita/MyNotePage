    # Nginx

    ここは、Nginxのことを書いているよ

    # Nginxでのパフォーマンスについて

    Webサーバで通信のパフォーマンスを上げるなら、キャッシュをするかGZIPをすればいいよ

    # キャッシュをする方法

    画像とかPDFやフォントをキャッシュしたいなら、どこかのconfファイルにしたの設定をする

    server {
  location ~ .*\.(jpe?g|gif|pdf|png|ico|swf|inc|svg|zip|eot|otf|ttf|woff|woff2) {
        expires 1d;
        access_log off;
    		open_file_cache max=100 inactive=10s;
  }
}

    # expires

    expiresはクライアントでキャッシュする期間を指しているよ

    # access_log

    access_logはアクセスログに記録するかを指しているよ

    # open_file_cache

    open_file_cacheは、maxとinactiveがあるよ

    # max

    maxは、キャッシュが格納できるエントリ数を決めていて、キャッシュのファイル数がmaxを超えたら古いファイルから消える

    # Inactive

    キャッシュエントリを格納しておく秒数で、簡単にいうとアクセスがないキャッシュの有効期限のこと。キャッシュエントリーにアクセスがあると最初から有効期限が設定される。

    # GZIPで圧縮する方法

    GZIP

    server {
	location {
    gzip on;
		gzip_types application/html image/jpeg;
    gzip_static always;
  }
}

    # gzip

    gzip onを書くとgzipを使うことになるよ

    # gzip_types

    gipしたいコンテンツタイプを指定するよ

    # gzip_static

    gzip_staticは、毎回圧縮するのはサーバの負荷がかかるので、１度作ったgzipがあったらそれを使うか決めるディレクティブ
