    #　Kotlinの環境構築について

    # Kotlin in docker

    Kotlin、Docker

    絶対にJavaをintallしたくないけどdockerでKotlinする

    # Kotlinのdocker-compose.yml

    Kotlinのyamlファイルを作る

    version: '3'
services:
  kotlin:
    image: zenika/kotlin:latest
    container_name: kotlin_container
    working_dir: /app
    volumes:
      - .:/app

    # Makefile

    コマンドを覚えるのが大変なのでMakefileを作る

    
.PHONY:up
up:
	docker-compose up -d --build

.PHONY:down
down:
	docker-compose down

.PHONY:compile
compile:
	docker-compose run kotlin kotlinc ${TARGET}

.PHONY:exec
exec:
	docker-compose run kotlin kotlin ${TARGET}

    # make up

    KotlinのDocker containerを立ち上げるよ

    # make down

    立ち上げたKotlinのDocker containerを終わらせるよ

    # make compile TARGET=

    TARGETで指定したファイルパスの.ktファイルをcompileする

    # make exec TARGET=

    TARGETで指定したファイルパスの実行ファイルを実行する

    ○○○○.classの○○○○の部分をTARGETファイル名になるよ
